236     # Returns a friendlier version of the identifier. Set skip_first_part to 
237     # omit the first part of the identifier (typically a region name) where
 
238     # there is more than one part.
 
239     #
 
240     # For example:
 
241     #
 
242     #   Timezone.get('Europe/Paris').friendly_identifier(false)          #=> "Europe - Paris"
 
243     #   Timezone.get('Europe/Paris').friendly_identifier(true)           #=> "Paris"
 
244     #   Timezone.get('America/Indiana/Knox').friendly_identifier(false)  #=> "America - Knox, Indiana"
 
245     #   Timezone.get('America/Indiana/Knox').friendly_identifier(true)   #=> "Knox, Indiana"           
246     def friendly_identifier(skip_first_part = false)
 
247       parts = identifier.split('/')
 
248       if parts.empty?
 
249         # shouldn't happen
 
250         identifier
 
251       elsif parts.length == 1        
252         parts[0]
 
253       else
 
254         if skip_first_part
 
255           result = ''
 
256         else
 
257           result = parts[0] + ' - '
 
258         end
 
259         
260         parts[1, parts.length - 1].reverse_each {|part|
 
261           part.gsub!(/_/, ' ')
 
262           
263           if part.index(/[a-z]/)
 
264             # Missing a space if a lower case followed by an upper case and the
 
265             # name isn't McXxxx.

297     # Returns the TimezonePeriod for the given local time. local can either be
 
298     # a DateTime, Time or integer timestamp (Time.to_i). Any timezone 
299     # information in local is ignored (it is treated as a time in the current 
300     # timezone).
 
301     #
 
302     # Warning: There are local times that have no equivalent UTC times (e.g.
 
303     # in the transition from standard time to daylight savings time). There are
 
304     # also local times that have more than one UTC equivalent (e.g. in the
 
305     # transition from daylight savings time to standard time).
 
306     #
 
307     # In the first case (no equivalent UTC time), a PeriodNotFound exception
 
308     # will be raised.
 
309     #
 
310     # In the second case (more than one equivalent UTC time), an AmbiguousTime
 
311     # exception will be raised unless the optional dst parameter or block
 
312     # handles the ambiguity. 
313     #
 
314     # If the ambiguity is due to a transition from daylight savings time to
 
315     # standard time, the dst parameter can be used to select whether the 
316     # daylight savings time or local time is used. For example,
 
317     #
 
318     #   Timezone.get('America/New_York').period_for_local(DateTime.new(2004,10,31,1,30,0))
 
319     #
 
320     # would raise an AmbiguousTime exception.
 
321     #
 
322     # Specifying dst=true would the daylight savings period from April to 
323     # October 2004. Specifying dst=false would return the standard period
 
324     # from October 2004 to April 2005.
 
325     #
 
326     # If the dst parameter does not resolve the ambiguity, and a block is 
327     # specified, it is called. The block must take a single parameter - an
 
328     # array of the periods that need to be resolved. The block can select and
 
329     # return a single period or return nil or an empty array
 
330     # to cause an AmbiguousTime exception to be raised.
 
331     def period_for_local(local, dst = nil)            
332       results = periods_for_local(local)
 
333       
334       if results.empty?
 
335         raise PeriodNotFound
 
336       elsif results.size < 2
 
337         results.first
 
338       else
 
339         # ambiguous result try to resolve
 
340         
341         if !dst.nil?
 
342           matches = results.find_all {|period| period.dst? == dst}
 
343           results = matches if !matches.empty?            
344         end
 
345         
346         if results.size < 2
 
347           results.first
 
348         else
 
349           # still ambiguous, try the block
 
350                     
351           if block_given?
 
352             results = yield results
 
353           end
 
354           
355           if results.is_a?(TimezonePeriod)
 
356             results
 
357           elsif results && results.size == 1
 
358             results.first
 
359           else          
360             raise AmbiguousTime, "#{local} is an ambiguous local time."
 
361           end
 
362         end
 
363       end      
364     end

383     # also local times that have more than one UTC equivalent (e.g. in the
 
384     # transition from daylight savings time to standard time).
 
385     #
 
386     # In the first case (no equivalent UTC time), a PeriodNotFound exception
 
387     # will be raised.

389     # In the second case (more than one equivalent UTC time), an AmbiguousTime
 
390     # exception will be raised unless the optional dst parameter or block
 
391     # handles the ambiguity.

393     # If the ambiguity is due to a transition from daylight savings time to
 
394     # standard time, the dst parameter can be used to select whether the

396     #
 
397     #   Timezone.get('America/New_York').local_to_utc(DateTime.new(2004,10,31,1,30,0))

399     # would raise an AmbiguousTime exception.

401     # Specifying dst=true would return 2004-10-31 5:30:00. Specifying dst=false

404     # If the dst parameter does not resolve the ambiguity, and a block is

408     # to cause an AmbiguousTime exception to be raised.
 
409     def local_to_utc(local, dst = nil)
 
410       TimeOrDateTime.wrap(local) {|wrapped|
 
411         if block_given?
 
412           period = period_for_local(wrapped, dst) {|periods| yield periods }
 
413         else
 
414           period = period_for_local(wrapped, dst)
 
415         end
 
416         
417         period.to_utc(wrapped)
 
418       }
 
419     end

441     # Converts a time in UTC to local time and returns it as a string 
442     # according to the given format. The formatting is identical to 
443     # Time.strftime and DateTime.strftime, except %Z is replaced with the
 
444     # timezone abbreviation for the specified time (for example, EST or EDT).        
445     def strftime(format, utc = Time.now.utc)      
446       period = period_for_utc(utc)
 
447       local = period.to_local(utc)      
448       local = Time.at(local).utc unless local.kind_of?(Time) || local.kind_of?(DateTime)

458       end
