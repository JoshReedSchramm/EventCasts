4   # AutosaveAssociation is a module that takes care of automatically saving
 
5   # your associations when the parent is saved. In addition to saving, it
 
6   # also destroys any associations that were marked for destruction.
 
7   # (See mark_for_destruction and marked_for_destruction?)
 
8   #
 
9   # Saving of the parent, its associations, and the destruction of marked
 
10   # associations, all happen inside 1 transaction. This should never leave the
 
11   # database in an inconsistent state after, for instance, mass assigning
 
12   # attributes and saving them.
 
13   #
 
14   # If validations for any of the associations fail, their error messages will
 
15   # be applied to the parent.
 
16   #
 
17   # Note that it also means that associations marked for destruction won't
 
18   # be destroyed directly. They will however still be marked for destruction.
 
19   #
 
20   # === One-to-one Example
 
21   #
 
22   # Consider a Post model with one Author:
 
23   #
 
24   #   class Post
 
25   #     has_one :author, :autosave => true
 
26   #   end
 
27   #
 
28   # Saving changes to the parent and its associated model can now be performed
 
29   # automatically _and_ atomically:
 
30   #
 
31   #   post = Post.find(1)
 
32   #   post.title # => "The current global position of migrating ducks"
 
33   #   post.author.name # => "alloy"
 
34   #
 
35   #   post.title = "On the migration of ducks"
 
36   #   post.author.name = "Eloy Duran"
 
37   #
 
38   #   post.save
 
39   #   post.reload
 
40   #   post.title # => "On the migration of ducks"
 
41   #   post.author.name # => "Eloy Duran"
 
42   #
 
43   # Destroying an associated model, as part of the parent's save action, is as
 
44   # simple as marking it for destruction:
 
45   #
 
46   #   post.author.mark_for_destruction
 
47   #   post.author.marked_for_destruction? # => true
 
48   #
 
49   # Note that the model is _not_ yet removed from the database:
 
50   #   id = post.author.id
 
51   #   Author.find_by_id(id).nil? # => false
 
52   #
 
53   #   post.save
 
54   #   post.reload.author # => nil
 
55   #
 
56   # Now it _is_ removed from the database:
 
57   #   Author.find_by_id(id).nil? # => true
 
58   #
 
59   # === One-to-many Example
 
60   #
 
61   # Consider a Post model with many Comments:
 
62   #
 
63   #   class Post
 
64   #     has_many :comments, :autosave => true
 
65   #   end
 
66   #
 
67   # Saving changes to the parent and its associated model can now be performed
 
68   # automatically _and_ atomically:
 
69   #
 
70   #   post = Post.find(1)
 
71   #   post.title # => "The current global position of migrating ducks"
 
72   #   post.comments.first.body # => "Wow, awesome info thanks!"
 
73   #   post.comments.last.body # => "Actually, your article should be named differently."
 
74   #
 
75   #   post.title = "On the migration of ducks"
 
76   #   post.comments.last.body = "Actually, your article should be named differently. [UPDATED]: You are right, thanks."
 
77   #
 
78   #   post.save
 
79   #   post.reload
 
80   #   post.title # => "On the migration of ducks"
 
81   #   post.comments.last.body # => "Actually, your article should be named differently. [UPDATED]: You are right, thanks."
 
82   #
 
83   # Destroying one of the associated models members, as part of the parent's
 
84   # save action, is as simple as marking it for destruction:
 
85   #
 
86   #   post.comments.last.mark_for_destruction
 
87   #   post.comments.last.marked_for_destruction? # => true
 
88   #   post.comments.length # => 2
 
89   #
 
90   # Note that the model is _not_ yet removed from the database:
 
91   #   id = post.comments.last.id
 
92   #   Comment.find_by_id(id).nil? # => false
 
93   #
 
94   #   post.save
 
95   #   post.reload.comments.length # => 1
 
96   #
 
97   # Now it _is_ removed from the database:
 
98   #   Comment.find_by_id(id).nil? # => true

103   # enabled associations. If any of the associations fail validation, its

156       # Adds a validate and save callback for the association as specified by
 
157       # the +reflection+.
 
158       #
 
159       # For performance reasons, we don't check whether to validate at runtime,
 
160       # but instead only define the method and callback when needed. However,
 
161       # this can change, for instance, when using nested attributes, which is
 
162       # called _after_ the association has been defined. Since we don't want
 
163       # the callbacks to get defined multiple times, there are guards that
 
164       # check if the save or validation methods have already been defined
 
165       # before actually defining them.
 
166       def add_autosave_association_callbacks(reflection)
 
167         save_method = :"autosave_associated_records_for_#{reflection.name}"
 
168         validation_method = :"validate_associated_records_for_#{reflection.name}"
 
169         collection = reflection.collection?
 
170 
171         unless method_defined?(save_method)
 
172           if collection
 
173             before_save :before_save_collection_association
 
174 
175             define_method(save_method) { save_collection_association(reflection) }
 
176             # Doesn't use after_save as that would save associations added in after_create/after_update twice
 
177             after_create save_method
 
178             after_update save_method
 
179           else
 
180             if reflection.macro == :has_one
 
181               define_method(save_method) { save_has_one_association(reflection) }
 
182               after_save save_method
 
183             else
 
184               define_method(save_method) { save_belongs_to_association(reflection) }
 
185               before_save save_method
 
186             end
 
187           end
 
188         end
 
189 
190         if reflection.validate? && !method_defined?(validation_method)
 
191           method = (collection ? :validate_collection_association : :validate_single_association)
 
192           define_method(validation_method) { send(method, reflection) }
 
193           validate validation_method
 
194         end
 
195       end

227     # Returns the record for an association collection that should be validated
 
228     # or saved. If +autosave+ is +false+ only new records will be returned,
 
229     # unless the parent is/was a new record itself.
 
230     def associated_records_to_validate_or_save(association, new_record, autosave)
 
231       if new_record
 
232         association
 
233       elsif autosave
 
234         association.target.find_all { |record| record.changed_for_autosave? }
 
235       else
 
236         association.target.find_all { |record| record.new_record? }

295     # Saves any new associated records, or all loaded autosave associations if

298     # In addition, it destroys all children that were marked for destruction
 
299     # with mark_for_destruction.

301     # This all happens inside a transaction, _if_ the Transactions module is included into
 
302     # ActiveRecord::Base after the AutosaveAssociation module, which it does by default.
 
303     def save_collection_association(reflection)
 
304       if association = association_instance_get(reflection.name)
 
305         autosave = reflection.options[:autosave]
 
306 
307         if records = associated_records_to_validate_or_save(association, @new_record_before_save, autosave)
 
308           records.each do |record|
 
309             next if record.destroyed?
 
310 
311             if autosave && record.marked_for_destruction?
 
312               association.destroy(record)
 
313             elsif autosave != false && (@new_record_before_save || record.new_record?)
 
314               if autosave
 
315                 saved = association.send(:insert_record, record, false, false)
 
316               else
 
317                 association.send(:insert_record, record)
 
318               end
 
319             elsif autosave

321             end
