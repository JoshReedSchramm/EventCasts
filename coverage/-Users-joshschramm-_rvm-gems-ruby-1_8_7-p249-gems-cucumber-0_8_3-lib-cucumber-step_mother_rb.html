285     def best_matches(step_name, step_matches) #:nodoc:
 
286       no_groups      = step_matches.select {|step_match| step_match.args.length == 0}
 
287       max_arg_length = step_matches.map {|step_match| step_match.args.length }.max
 
288       top_groups     = step_matches.select {|step_match| step_match.args.length == max_arg_length }
 
289 
290       if no_groups.any?
 
291         longest_regexp_length = no_groups.map {|step_match| step_match.text_length }.max
 
292         no_groups.select {|step_match| step_match.text_length == longest_regexp_length }
 
293       elsif top_groups.any?
 
294         shortest_capture_length = top_groups.map {|step_match| step_match.args.inject(0) {|sum, c| sum + c.to_s.length } }.min
 
295         top_groups.select {|step_match| step_match.args.inject(0) {|sum, c| sum + c.to_s.length } == shortest_capture_length }
