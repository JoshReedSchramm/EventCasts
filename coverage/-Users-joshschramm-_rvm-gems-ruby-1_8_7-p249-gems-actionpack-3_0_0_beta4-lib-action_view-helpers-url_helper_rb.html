38       # instead of the fully qualified URL like "http://example.com/controller/action".

40       # When called from a view, +url_for+ returns an HTML escaped url. If you

43       # ==== Options
 
44       # * <tt>:anchor</tt> - Specifies the anchor name to be appended to the path.

527       # True if the current request URI was generated by the given +options+.
 
528       #
 
529       # ==== Examples
 
530       # Let's say we're in the <tt>/shop/checkout?order=desc</tt> action.
 
531       #
 
532       #   current_page?(:action => 'process')
 
533       #   # => false
 
534       #
 
535       #   current_page?(:controller => 'shop', :action => 'checkout')

542       #   # => true

544       #   current_page?(:controller => 'library', :action => 'checkout')
 
545       #   # => false
 
546       #
 
547       # Let's say we're in the <tt>/shop/checkout?order=desc&page=1</tt> action.
 
548       #
 
549       #   current_page?(:action => 'process')

552       #   current_page?(:controller => 'shop', :action => 'checkout')
 
553       #   # => true
 
554       #
 
555       #   current_page?(:controller => 'shop', :action => 'checkout', :order => 'desc', :page=>'1')
 
556       #   # => true
 
557       #
 
558       #   current_page?(:controller => 'shop', :action => 'checkout', :order => 'desc', :page=>'2')
 
559       #   # => false
 
560       #
 
561       #   current_page?(:controller => 'shop', :action => 'checkout', :order => 'desc')
 
562       #   # => false
 
563       #
 
564       #   current_page?(:action => 'checkout')
 
565       #   # => true
 
566       #
 
567       #   current_page?(:controller => 'library', :action => 'checkout')
 
568       #   # => false
 
569       def current_page?(options)
 
570         unless request
 
571           raise "You cannot use helpers that need to determine the current " \
 
572                 "page unless your view context provides a Request object " \
 
573                 "in a #request method"
 
574         end
 
575 
576         url_string = CGI.unescapeHTML(url_for(options))
 
577 
578         # We ignore any extra parameters in the request_uri if the
 
579         # submitted url doesn't have any either.  This lets the function
 
580         # work with things like ?order=asc
 
581         if url_string.index("?")
 
582           request_uri = request.fullpath
 
583         else
 
584           request_uri = request.path
 
585         end
 
586 
587         if url_string =~ /^\w+:\/\//
 
588           url_string == "#{request.protocol}#{request.host_with_port}#{request_uri}"
 
589         else
 
590           url_string == request_uri
 
591         end
 
592       end

642         # Processes the +html_options+ hash, converting the boolean
 
643         # attributes from true/false form into the form required by
 
644         # HTML/XHTML.  (An attribute is considered to be boolean if
 
645         # its name is listed in the given +bool_attrs+ array.)
 
646         #
 
647         # More specifically, for each boolean attribute in +html_options+
 
648         # given as:
 
649         #
 
650         #   "attr" => bool_value
 
651         #
 
652         # if the associated +bool_value+ evaluates to true, it is
 
653         # replaced with the attribute's name; otherwise the attribute is
 
654         # removed from the +html_options+ hash.  (See the XHTML 1.0 spec,
 
655         # section 4.5 "Attribute Minimization" for more:
 
656         # http://www.w3.org/TR/xhtml1/#h-4.5)
 
657         #
 
658         # Returns the updated +html_options+ hash, which is also modified
 
659         # in place.
 
660         #
 
661         # Example:
 
662         #
 
663         #   convert_boolean_attributes!( html_options,
 
664         #                                %w( checked disabled readonly ) )
 
665         def convert_boolean_attributes!(html_options, bool_attrs)
 
666           bool_attrs.each { |x| html_options[x] = x if html_options.delete(x) }
 
667           html_options
 
668         end
