30     # A generic "counter updater" implementation, intended primarily to be
 
31     # used by increment_counter and decrement_counter, but which may also
 
32     # be useful on its own. It simply does a direct SQL update for the record
 
33     # with the given ID, altering the given hash of counters by the amount
 
34     # given by the corresponding value:
 
35     #
 
36     # ==== Parameters
 
37     #
 
38     # * +id+ - The id of the object you wish to update a counter on or an Array of ids.
 
39     # * +counters+ - An Array of Hashes containing the names of the fields
 
40     #   to update as keys and the amount to update the field by as values.
 
41     #
 
42     # ==== Examples
 
43     #
 
44     #   # For the Post with id of 5, decrement the comment_count by 1, and
 
45     #   # increment the action_count by 1
 
46     #   Post.update_counters 5, :comment_count => -1, :action_count => 1
 
47     #   # Executes the following SQL:
 
48     #   # UPDATE posts
 
49     #   #    SET comment_count = comment_count - 1,
 
50     #   #        action_count = action_count + 1
 
51     #   #  WHERE id = 5
 
52     #
 
53     #   # For the Posts with id of 10 and 15, increment the comment_count by 1
 
54     #   Post.update_counters [10, 15], :comment_count => 1
 
55     #   # Executes the following SQL:
 
56     #   # UPDATE posts
 
57     #   #    SET comment_count = comment_count + 1,
 
58     #   #  WHERE id IN (10, 15)
 
59     def update_counters(id, counters)
 
60       updates = counters.map do |counter_name, value|
 
61         operator = value < 0 ? '-' : '+'
 
62         quoted_column = connection.quote_column_name(counter_name)
 
63         "#{quoted_column} = COALESCE(#{quoted_column}, 0) #{operator} #{value.abs}"
 
64       end
 
65 
66       update_all(updates.join(', '), primary_key => id )
 
67     end
