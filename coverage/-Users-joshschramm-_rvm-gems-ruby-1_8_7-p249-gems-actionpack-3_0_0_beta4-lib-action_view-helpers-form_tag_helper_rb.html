536         def extra_tags_for_form(html_options)
 
537           case method = html_options.delete("method").to_s
 
538             when /^get$/i # must be case-insentive, but can't use downcase as might be nil
 
539               html_options["method"] = "get"
 
540               ''
 
541             when /^post$/i, "", nil
 
542               html_options["method"] = "post"
 
543               protect_against_forgery? ? content_tag(:div, token_tag, :style => 'margin:0;padding:0;display:inline') : ''
 
544             else
 
545               html_options["method"] = "post"
 
546               content_tag(:div, tag(:input, :type => "hidden", :name => "_method", :value => method) + token_tag, :style => 'margin:0;padding:0;display:inline')
 
547           end
 
548         end
